47# Parses playlist json and gives DataFrame

53 playlist_dict = {} 

54 count = 0 # track order of songs in playlist 

55 

56 # Return empty playlist is empty

60 # Loop through playlist items 

61 for item in playlist["tracks"]["items"]: 

62 track = item["track"]

64 # Make sure item is track before parsing 

65 if track is not None:

71 playlist_dict[count] = {"Name": track["album"]["name"],
